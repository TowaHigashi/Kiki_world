<template>
    <div ref="container"></div>
</template>

<script setup>
import { onMounted, onUnmounted, ref } from 'vue' // onUnmountedを追加
import * as THREE from 'three'
import { OrbitControls } from 'three/examples/jsm/controls/OrbitControls'

const container = ref(null)  // HTML要素への参照
const raycaster = new THREE.Raycaster()
const mouse = new THREE.Vector2()
const buildings = []

let targetPosition = new THREE.Vector3() // カメラの移動先 3-3
let targetLookAt = new THREE.Vector3() // カメラの見る位置 3-3
let isMoving = false // カメラが移動中か 3-3

onMounted(() => {
    const scene = new THREE.Scene()
    scene.background = new THREE.Color(0xa0d8f0)

    const camera = new THREE.PerspectiveCamera(
        75,
        window.innerWidth / window.innerHeight,
        0.1,
        1000
    )

    const renderer = new THREE.WebGLRenderer({ antialias: true })
    renderer.setPixelRatio(window.devicePixelRatio)
    renderer.setSize(window.innerWidth, window.innerHeight)
    container.value.appendChild(renderer.domElement)

    // ===== ウィンドウサイズ変更の対応 =====
    const onResize = () => {
        camera.aspect = window.innerWidth / window.innerHeight
        camera.updateProjectionMatrix()
        renderer.setSize(window.innerWidth, window.innerHeight)
    }
    window.addEventListener('resize', onResize)

    // ===== クリック処理 (ドラッグ判定を追加) =====
    let mouseStartX = 0
    let mouseStartY = 0
    const onMouseDown = (event) => {
        mouseStartX = event.clientX
        mouseStartY = event.clientY
    }
    window.addEventListener('mousedown', onMouseDown)

    const onSceneClick = (event) => {
        // ドラッグした距離をチェック (5px以上動いていたらドラッグとみなして無視)
        const diffX = Math.abs(event.clientX - mouseStartX)
        const diffY = Math.abs(event.clientY - mouseStartY)
        if (diffX > 5 || diffY > 5) return

        mouse.x = (event.clientX / window.innerWidth) * 2 - 1
        mouse.y = -(event.clientY / window.innerHeight) * 2 + 1

        raycaster.setFromCamera(mouse, camera)
        const intersects = raycaster.intersectObjects(buildings)

        if (intersects.length > 0) {
            const selected = intersects[0].object
            // console.log('clicked:', intersects[0].object)
            targetLookAt.copy(selected.position) // 見る位置（建物そのもの）3-3

            targetPosition.set( // カメラの移動先（建物から少し離す）3-3
                selected.position.x + 3,
                selected.position.y + 2,
                selected.position.z + 3
            )

            isMoving = true

            console.log('targetPosition:', targetPosition)
            console.log('targetLookAt:', targetLookAt)
        }
    }
    window.addEventListener('click', onSceneClick)

    // 右クリックメニューを有効化する処理 
    renderer.domElement.addEventListener('contextmenu', (e) => { e.stopPropagation() }, true)

    // ===== 地面 =====
    const groundGeometry = new THREE.PlaneGeometry(8, 8)
    const groundMaterial = new THREE.MeshStandardMaterial({
        color: 0x228B22,
        side: THREE.DoubleSide
    })
    const ground = new THREE.Mesh(groundGeometry, groundMaterial)
    ground.rotation.x = Math.PI / 2
    scene.add(ground)

    // ===== 建物 =====
    const createBuilding = (x, z, width, depth, height) => {
        const geometry = new THREE.BoxGeometry(width, height, depth)
        const material = new THREE.MeshStandardMaterial({ color: 0xffffff })
        const building = new THREE.Mesh(geometry, material)

        building.position.set(x, height / 2, z)
        scene.add(building)

        const edges = new THREE.EdgesGeometry(geometry)
        const lineMaterial = new THREE.LineBasicMaterial({ color: 0x000000 })
        const line = new THREE.LineSegments(edges, lineMaterial)

        // 輪郭
        line.position.copy(building.position)
        scene.add(line)

        buildings.push(building)
    }

    createBuilding(-2, -1.7, 1.4, 1.4, 4.2)
    createBuilding(1.25, -1.8, 2.5, 1.5, 3)
    createBuilding(-1.3, 1, 2, 2.4, 1.8)
    createBuilding(1.6, 1, 2.1, 2, 2.5)

    // ===== ライト =====
    const ambientLight = new THREE.AmbientLight(0xffffff, 1)
    scene.add(ambientLight)

    const directionalLight = new THREE.DirectionalLight(0xffffff, 1)
    directionalLight.position.set(5, 10, 5)
    scene.add(directionalLight)

    // ===== カメラ位置 =====
    camera.position.set(5, 5, 5)
    camera.lookAt(0, 0, 0)

    // ===== マウス操作 =====
    const controls = new OrbitControls(camera, renderer.domElement)
    controls.enableDamping = true
    controls.enablePan = false // 追加
    controls.dampingFactor = 0.05
    controls.screenSpacePanning = false
    controls.minDistance = 1
    controls.maxDistance = 30
    controls.mouseButtons = {
        LEFT: THREE.MOUSE.ROTATE,
        MIDDLE: THREE.MOUSE.DOLLY,
        RIGHT: undefined // 右クリックメニューのためにundefinedを設定
    }

    // ドラッグ操作を開始したら自動移動を止める
    controls.addEventListener('start', () => {
        isMoving = false
    })

    // ===== アニメーション =====
    let animationId
    const animate = () => {
        animationId = requestAnimationFrame(animate)

        if (isMoving) {
            camera.position.lerp(targetPosition, 0.1)
            controls.target.lerp(targetLookAt, 0.1)

            if (camera.position.distanceTo(targetPosition) < 0.08) { // ビタッと止まるかどうか 3-3
                isMoving = false
            }
        }

        controls.update()
        renderer.render(scene, camera)
    }
    animate()

    // ===== クリーンアップ処理 =====
    onUnmounted(() => {
        window.removeEventListener('resize', onResize)
        window.removeEventListener('mousedown', onMouseDown) // 追加
        window.removeEventListener('click', onSceneClick)
        cancelAnimationFrame(animationId)
        controls.dispose()
        renderer.dispose() // シーン内のジオメトリやマテリアルも必要に応じてdisposeすると完璧です
    })
})
</script>

<script setup>
import { ref, onMounted, onUnmounted } from 'vue'

// 非表示状態の管理
const isVisible = ref(true)

const hidePage = () => {
    isVisible.value = false
}

onMounted(() => {
    // 画面のどこをクリックしても消えるように、windowにイベントを登録
    // { once: true } を付けることで、1回実行されたら自動で解除されます
    window.addEventListener('click', hidePage, { once: true })
})

onUnmounted(() => {
    // クリーンアップ
    window.removeEventListener('click', hidePage)
})
</script>

<template>
    <!-- v-if で表示・非表示を切り替え -->
    <div v-if="isVisible" class="start-overlay">
        <h1>Start</h1>
    </div>
</template>

<style scoped>
.start-overlay {
    /* 画面全体を覆い、他の要素（Three.jsなど）より上に表示 */
    position: fixed;
    top: 0;
    left: 0;
    width: 100%;
    height: 100%;
    background-color: transparent; /* 完全に透明にして操作を邪魔しない */
    z-index: 1000;
    display: flex;
    justify-content: center;
    align-items: center;

    /* ★ ここが重要：マウス操作を背後にスルーさせる */
    pointer-events: none;
}

h1 {
    font-size: 80px;
    color: white;
    margin: 0;
    pointer-events: none; /* 文字そのものへのクリック判定を無効化 */
    user-select: none; /* 文字を選択できないようにする */
}
</style>
